function get_author() {
  var author_str = "" ;
  var authors = new Array() ;
  parts = document.getElementsByTagName("META")  ;    
  for( head_part_idx = 0; head_part_idx < parts.length ; head_part_idx ++ ) {
    try {
      nodename = parts[ head_part_idx ].nodeName ;
      meta_name = parts[ head_part_idx ].attributes[ 'name' ].nodeValue ;
      if( meta_name == 'author' ) {
        meta_val = parts[ head_part_idx ].attributes[ 'content' ].nodeValue ;
        authors.push( meta_val ) ;
      }
    } catch( e ) { do_nothing= 0 ;} ;
  }
  author_str = authors.join( ", " ) ;

  return author_str ;
}
